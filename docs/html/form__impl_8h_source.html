44  public:
45  mouse_track() : mouse_tracking_(false) {}

47  void on_mouse_move(HWND hWnd) {
48  if (!mouse_tracking_) {
49  // Enable mouse tracking
50  TRACKMOUSEEVENT tme;
51  tme.cbSize = sizeof(tme);
52  tme.hwndTrack = hWnd;
53  tme.dwFlags = TME_HOVER | TME_LEAVE;
54  tme.dwHoverTime = HOVER_DEFAULT;
55  TrackMouseEvent(&tme);
56  mouse_tracking_ = true;
57  }
58  }
59 
60  void reset(HWND hWnd) { mouse_tracking_ = false; }
61  };
