188  static inline void position_h_scrollbar(const D2D1_RECT_F& rectA,
189  const D2D1_RECT_F& rectB, const D2D1_RECT_F& rectC, D2D1_RECT_F& rectD) {
190  const float A_width = rectA.right - rectA.left;
191  const float B_width = rectB.right - rectB.left;
192  const float C_width = rectC.right - rectC.left;
193 
194  // if both A and B have 0.f width set to 1.f
195  // if A has width 0.f but not B set to 0.f
196  const float width_ratio = A_width == 0.f ? (B_width == 0.f ? 1.f : 0.f) : B_width / A_width;
197 
198  const float width_d = C_width * width_ratio;
199  rectD.left = 0.f;
200  rectD.right = rectD.left + width_d;
201  rectD.top = rectC.top;
202  rectD.bottom = rectC.bottom;
203 
204  // compute how far B is along A
205  auto perc_b_along_a = [&]() {
206  const float start_pos = rectA.left + ((rectB.right - rectB.left) / 2.f);
207  const float end_pos = rectA.right - ((rectB.right - rectB.left) / 2.f);
208  const float center_B = rectB.left + ((rectB.right - rectB.left) / 2.f);
209 
210  if (end_pos == start_pos)
211  return 100.f;
212  else
213  return 100.f * (center_B - start_pos) / (end_pos - start_pos);
214  };
215 
216  // position C within D
217  pos_rect(rectC, rectD, perc_b_along_a(), 0.f);
218  }

221  static inline void position_v_scrollbar(const D2D1_RECT_F& rectA,
222  const D2D1_RECT_F& rectB, const D2D1_RECT_F& rectC, D2D1_RECT_F& rectD) {
223  const float A_height = rectA.bottom - rectA.top;
224  const float B_height = rectB.bottom - rectB.top;
225  const float C_height = rectC.bottom - rectC.top;
226 
227  // if both A and B have 0.f height set to 1.f
228  // if A has height 0.f but not B set to 0.f
229  const float height_ratio = A_height == 0.f ? (B_height == 0.f ? 1.f : 0.f) : B_height / A_height;
230 
231  const float height_d = C_height * height_ratio;
232  rectD.top = 0.f;
233  rectD.bottom = rectD.top + height_d;
234  rectD.left = rectC.left;
235  rectD.right = rectC.right;
236 
237  // compute how far B is along A
238  auto perc_b_along_a = [&]() {
239  const float start_pos = rectA.top + ((rectB.bottom - rectB.top) / 2.f);
240  const float end_pos = rectA.bottom - ((rectB.bottom - rectB.top) / 2.f);
241  const float center_B = rectB.top + ((rectB.bottom - rectB.top) / 2.f);
242 
243  if (end_pos == start_pos)
244  return 100.f;
245  else
246  return 100.f * (center_B - start_pos) / (end_pos - start_pos);

291  static double tod(const double& d, int precision) {
292  int y = (int)d;
293  double z = d - (double)y;
294  double m = pow(10, precision);
295  double q = z * m;
296  double r = round(q);
297 
298  return static_cast<double>(y) + (1.0 / m) * r;
299  }

390  static inline void fit_rect(const D2D1_RECT_F rect_container, D2D1_RECT_F& rect,
391  bool enlarge_if_smaller, bool keep_aspect_ratio, bool center) {
392  auto width = rect_container.right - rect_container.left;
393  auto height = rect_container.bottom - rect_container.top;
394 
395  // deduce old height and ratio
396  const auto old_height = rect.bottom - rect.top;
397  const auto old_width = rect.right - rect.left;
398  const auto ratio = old_width / old_height;
399 
400  if (enlarge_if_smaller == false) {
401  if (old_width < width && old_height < height) {
402  // both sides of the rectangle are smaller than the container, preserve size
403  width = old_width;
404  height = old_height;
405  }
406  }
407 
408  // save target dimensions
409  const auto control_w = width;
410  const auto control_h = height;
411 
412  if (ratio == 1) {
413  if (keep_aspect_ratio) {
414  if (width > height)
415  width = height; // landscape
416  else
417  height = width; // portrait
418  }
419  }
420  else {
421  if (keep_aspect_ratio) {
422  // adjust either new width or height to keep aspect ratio
423  if (old_width > old_height) {
424  // old width is greater than old height
425  // adjust new height using new width to keep aspect ratio
426  height = width / ratio;
427 
428  if (height > control_h) {
429  // new width is greater than target width, adjust it accordingly
430  height = control_h;
431  width = height * ratio;
432  }
433  }
434  else {
435  // old height is greater than old width
436  // adjust new width using new height to keep aspect ratio
437  width = height * ratio;
438 
439  if (width > control_w) {
440  // new width is greater than target width, adjust it accordingly
441  width = control_w;
442  height = width / ratio;
443  }
444  }
445  }
446  }
447 
448  rect.left = 0.f;
449  rect.top = 0.f;
450 
451  if (center) {
452  rect.left = ((rect_container.right - rect_container.left) - width) / 2.f;
453  rect.top = ((rect_container.bottom - rect_container.top) - height) / 2.f;
454  }
455 
456  rect.left += rect_container.left;
457  rect.top += rect_container.top;
458  rect.right = rect.left + width;
459  rect.bottom = rect.top + height;
460  }

462  static inline void create_bitmap(ID2D1RenderTarget* pRenderTarget,
463  IWICImagingFactory* pIWICFactory, IWICBitmapDecoder* pDecoder,
464  IWICBitmapFrameDecode* pSource, IWICFormatConverter* pConverter,
465  ID2D1Bitmap** ppBitmap, size target_size,
466  bool enlarge_if_smaller, bool keep_aspect_ratio, image_quality quality) {
467  IWICBitmapScaler* pScaler = NULL;
468 
469  // Create the initial frame.
470  HRESULT hr = pDecoder->GetFrame(0, &pSource);
471 
472  if (SUCCEEDED(hr)) {
473  // Convert the image format to 32bppPBGRA
474  // (DXGI_FORMAT_B8G8R8A8_UNORM + D2D1_ALPHA_MODE_PREMULTIPLIED).
475  hr = pIWICFactory->CreateFormatConverter(&pConverter);
476  }
477 
478  if (SUCCEEDED(hr)) {
479  UINT32 original_width = 0;
480  UINT32 original_height = 0;
481  hr = pSource->GetSize(&original_width, &original_height);
482  if (SUCCEEDED(hr)) {
483  D2D1_RECT_F rect_container = { 0, 0, 0, 0 };
484  rect_container.right = rect_container.left + target_size.width;
485  rect_container.bottom = rect_container.top + target_size.height;
486  D2D1_RECT_F rect = { 0, 0, 0, 0 };
487  rect.right = rect.left + static_cast<float>(original_width);
488  rect.bottom = rect.top + static_cast<float>(original_height);
489  fit_rect(rect_container, rect, enlarge_if_smaller, keep_aspect_ratio, false);
490  auto new_width = static_cast<UINT>(rect.right - rect.left);
491  auto new_height = static_cast<UINT>(rect.bottom - rect.top);
492 
493  if (new_width != original_width || new_height != original_height) {
494  // scale image
495  hr = pIWICFactory->CreateBitmapScaler(&pScaler);
496 
497  if (SUCCEEDED(hr)) {
498  WICBitmapInterpolationMode mode;
499  switch (quality) {
500  case image_quality::low:
501  mode = WICBitmapInterpolationModeNearestNeighbor;
502  break;
503  case image_quality::high:
504  mode = WICBitmapInterpolationModeHighQualityCubic;
505  break;
506  case image_quality::medium:
507  default:
508  mode = WICBitmapInterpolationModeLinear;
509  break;
510  }
511  hr = pScaler->Initialize(pSource, new_width, new_height, mode);
512  }
513  if (SUCCEEDED(hr)) {
514  hr = pConverter->Initialize(pScaler, GUID_WICPixelFormat32bppPBGRA,
515  WICBitmapDitherTypeNone, NULL, 0.f, WICBitmapPaletteTypeMedianCut);
516  }
517  }
518  else {
519  // don't scale image
520  hr = pConverter->Initialize(pSource, GUID_WICPixelFormat32bppPBGRA,
521  WICBitmapDitherTypeNone, NULL, 0.f, WICBitmapPaletteTypeMedianCut);
522  }
523  }
524  }
525 
526  if (SUCCEEDED(hr)) {
527  // Create a Direct2D bitmap from the WIC bitmap.
528  hr = pRenderTarget->CreateBitmapFromWicBitmap(pConverter, NULL, ppBitmap);
529  }
530 
531  safe_release(&pScaler);
532  }

534  static inline HRESULT load_bitmap_resource(ID2D1RenderTarget* pRenderTarget,
535  IWICImagingFactory* pIWICFactory, HINSTANCE hInst, int id_image,
536  std::string resource_type, ID2D1Bitmap** ppBitmap,
537  size target_size, bool enlarge_if_smaller, bool keep_aspect_ratio, image_quality quality) {
538  IWICBitmapDecoder* pDecoder = NULL;
539  IWICBitmapFrameDecode* pSource = NULL;
540  IWICStream* pStream = NULL;
541  IWICFormatConverter* pConverter = NULL;
542 
543  HRSRC imageResHandle = NULL;
544  HGLOBAL imageResDataHandle = NULL;
545  void* pImageFile = NULL;
546  DWORD imageFileSize = 0;
547 
548  // Locate the resource.
549  imageResHandle = FindResourceA(hInst, MAKEINTRESOURCEA(id_image), resource_type.c_str());
550  HRESULT hr = imageResHandle ? S_OK : E_FAIL;
551  if (SUCCEEDED(hr)) {
552  // Load the resource.
553  imageResDataHandle = LoadResource(hInst, imageResHandle);
554  hr = imageResDataHandle ? S_OK : E_FAIL;
555  }
556 
557  if (SUCCEEDED(hr)) {
558  // Lock it to get a system memory pointer.
559  pImageFile = LockResource(imageResDataHandle);
560  hr = pImageFile ? S_OK : E_FAIL;
561  }
562 
563  if (SUCCEEDED(hr)) {
564  // Calculate the size.
565  imageFileSize = SizeofResource(hInst, imageResHandle);
566  hr = imageFileSize ? S_OK : E_FAIL;
567  }
568  if (SUCCEEDED(hr)) {
569  // Create a WIC stream to map onto the memory.
570  hr = pIWICFactory->CreateStream(&pStream);
571  }
572  if (SUCCEEDED(hr)) {
573  // Initialize the stream with the memory pointer and size.
574  hr = pStream->InitializeFromMemory(reinterpret_cast<BYTE*>(pImageFile), imageFileSize);
575  }
576  if (SUCCEEDED(hr)) {
577  // Create a decoder for the stream.
578  hr = pIWICFactory->CreateDecoderFromStream(pStream, NULL,
579  WICDecodeMetadataCacheOnLoad, &pDecoder);
580  }
581 
582  if (SUCCEEDED(hr))
583  create_bitmap(pRenderTarget, pIWICFactory, pDecoder, pSource, pConverter,
584  ppBitmap, target_size, enlarge_if_smaller, keep_aspect_ratio, quality);
585 
586  safe_release(&pDecoder);
587  safe_release(&pSource);
588  safe_release(&pStream);
589  safe_release(&pConverter);
590  return hr;
591  }
