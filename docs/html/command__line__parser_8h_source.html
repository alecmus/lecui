30 class command_line_parser : public std::vector<char*> {
31 public:
32  command_line_parser() {
33  // Save local copy of the command line string, because
34  // ParseCmdLine() modifies this string while parsing it.

37  if (m_cmdline) {

39  ** customization by Alec Musasa on 26 Feb 2014

43  strcpy_s(m_cmdline, strlen(cmdline) + 1, cmdline);
44  ParseCmdLine();
45  }
46  }

68 private:
69  PSZ m_cmdline; // the command line string
70 
72  // Parse m_cmdline into individual tokens, which are delimited by spaces. If a
73  // token begins with a quote, then that token is terminated by the next quote
74  // followed immediately by a space or terminator. This allows tokens to contain
75  // spaces.
76  // This input string: This "is" a ""test"" "of the parsing" alg"o"rithm.
77  // Produces these tokens: This, is, a, "test", of the parsing, alg"o"rithm
79  void ParseCmdLine() {
80  enum {
81  TERM = '\0',
82  QUOTE = '\"'
83  };
84 
85  bool bInQuotes = false;
86  PSZ pargs = m_cmdline;
87 
88  while (*pargs) {
89  while (isspace(*pargs)) // skip leading whitespace
90  pargs++;
91 
92  bInQuotes = (*pargs == QUOTE); // see if this token is quoted
93 
94  if (bInQuotes) // skip leading quote
95  pargs++;
96 
97  push_back(pargs); // store position of current token
98 
99  // Find next token.
100  // NOTE: Args are normally terminated by whitespace, unless the
101  // arg is quoted. That's why we handle the two cases separately,
102  // even though they are very similar.
103  if (bInQuotes) {
104  // find next quote followed by a space or terminator
105  while (*pargs &&
106  !(*pargs == QUOTE && (isspace(pargs[1]) || pargs[1] == TERM)))
107  pargs++;
108  if (*pargs) {
109  *pargs = TERM; // terminate token
110  if (pargs[1]) // if quoted token not followed by a terminator
111  pargs += 2; // advance to next token
112  }
113  }
114  else {
115  // skip to next non-whitespace character
116  while (*pargs && !isspace(*pargs))
117  pargs++;
118  if (*pargs && isspace(*pargs)) { // end of token
119  *pargs = TERM; // terminate token
120  pargs++; // advance to next token or terminator
121  }
122  }
123  }
124  }
125 };
